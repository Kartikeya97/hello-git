q3 from tkinter import *
   root = Tk()
   root.title('A')
   root.configure(background='blue')
   def UserE:
   	userL = Label(root, text='User Name: ')
   	userL.pack()
   	userE = Entry(root)
   	userE.pack()

   retryB=Button(root, text='Retry', bg='green',\
                  width=25,command=UserE)    
   exitB=Button(root, text='exit', width=25, \
                 command=root.destroy)
   retryB.pack()
   exitB.pack()
   root.mainloop()
